تطوير APIs باستخدام Node.js

نُشر في 3 يونيو 2024 | بواسطة سلطان عايض

Node.js يتيح لنا بناء خوادم قوية وسريعة. سنتعلم كيفية إنشاء API بسيط باستخدام Express.js.

JavaScript 
const express = require('express');
const app = express();

app.use(express.json());

// GET endpoint
app.get('/api/users', (req, res) => {
  res.json([
    { id: 1, name: 'أحمد' },
    { id: 2, name: 'فاطمة' }
  ]);
});

// POST endpoint
app.post('/api/users', (req, res) => {
  const { name } = req.body;
  res.json({ message: `تم إنشاء المستخدم ${name} بنجاح` });
});

app.listen(3000, () => {
  console.log('Server running on port 3000');
});
